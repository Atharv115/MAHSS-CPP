//To find the largest and second largest in a matrix
//(same program for smallest and second smallest with change in comparison operator)
#include<iostream>
using namespace std;
int main() {
	int a[10][10], r, c, i, j, l1, l2;
	cout<<"Enter the number of rows in the matrix: ";
	cin>>r;
	cout<<"\nEnter the number of columns in the array: ";
	cin>>c;
	cout<<"\nEnter the elements in the matrix: ";
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			cin>>a[i][j];
		}
	}
	
	l1=a[0][0];
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			if(a[i][j]>l1) {
				l1=a[i][j];
			}
		}
	}
	
	if(l1==a[0][0]) {
		l2=a[0][1];
	}
	else {
		l2=a[0][0];
	}
	
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			if(a[i][j]>l2 && a[i][j]<l1) {
				l2=a[i][j];
			}
		}
	}
	
	cout<<"\nThe largest number is "<<l1<<" and the second largest number is "<<l2;
	
	
}


//To find the sum of odd elements and even elements in a matrix
#include<iostream>
using namespace std;
int main() {
	int a[10][10], r, c, i, j, sum_odd, sum_even;
	cout<<"Enter the number of rows in the matrix: ";
	cin>>r;
	cout<<"\nEnter the number of columns in the array: ";
	cin>>c;
	cout<<"\nEnter the elements in the matrix: ";
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			cin>>a[i][j];
		}
	}
	
	cout<<"\nThe matrix you entered:\n";
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			cout<<a[i][j];
		}
		cout<<"\n";
	}
	
	sum_odd=0;
	sum_even=0;
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			if(a[i][j]%2!=0) {
				sum_odd=sum_odd+a[i][j];
			}
			else {
				sum_even=sum_even+a[i][j];
			}
		}
	}
	
	cout<<"\nThe sum of odd numbers is "<<sum_odd<<" and the sum of even numbers is "<<sum_even;
	
	
}


//To search for an element in a matrix
#include<iostream>
using namespace std;
int main() {
	int a[10][10], r, c, i, j, count=0, x;
	cout<<"Enter the number of rows in the matrix: ";
	cin>>r;
	cout<<"\nEnter the number of columns in the array: ";
	cin>>c;
	cout<<"\nEnter the elements in the matrix: ";
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			cin>>a[i][j];
		}
	}
	
	cout<<"\nThe matrix you entered:\n";
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			cout<<a[i][j]<<"\t";
		}
		cout<<"\n";
	}
	
	cout<<"\nEnter the element to be searched: ";
	cin>>x;
	
	for(i=0;i<r;i++) {
		for(j=0;j<c;j++) {
			if(a[i][j]==x) {
				cout<<"\nElement present at index "<<i<<j;
				count++;
			}
		}
	}
	
	if(count==0) {
		cout<<"\nElement not present.";
	}
	
}

		for(j=0;j<c2;j++) {
			cin>>a2[i][j];
		}
	}
	
	cout<<"\nThe matrix you entered:\n";
	for(i=0;i<r2;i++) {
		for(j=0;j<c2;j++) {
			cout<<a2[i][j]<<"\t";
		}
		cout<<"\n";
	}
    if(c==r2) {
    	for(i=0;i<r;i++) {
    		for(j=0;j<c;j++) {
    			mul[i][j]=0;
    			for(k=0;k<c;k++) {
    				mul[i][j]=mul[i][j]+a[i][k]*a2[k][j];
				}
			}
		}
		cout<<"\nThe product of the matrices you entered:\n";
		for(i=0;i<r;i++) {
			for(j=0;j<c2;j++) {

	}
	
	else {

}
