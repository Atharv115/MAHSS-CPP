//Write a function in C++ to count the words 'to' and 'the' present in the file
//poem.txt
#include<iostream>
#include<fstream>
#include<string.h>
using namespace std;

int main() {
	ifstream a;
	int c1=0,c2=0;
	char ch[10];
	a.open("poem.txt");
	while(a) {
		a>>ch;
		if(strcmp(ch,"to")==0)
			c1++;
		if(strcmp(ch,"the")==0)
			c2++;
	}
	a.close();
	cout<<"Count of to: "<<c1<<"\nCount of the: "<<c2;
}


//To count and display the number of alphabets and number of spaces in an existing
//file lines.txt
#include<iostream>
#include<fstream>
#include<string.h>
using namespace std;

int main() {
	ifstream a;
	int c1=0,c2=0;
	char ch;
	a.open("line.txt");
	while(a) {
		a.get(ch);
		if(ch>='A'&&ch<='Z'||ch>='a'&&ch<='z')
			c1++;
		if(ch==' ')
			c2++;
	}
	a.close();
	cout<<"Count of alphabets: "<<c1<<"\nCount of spaces: "<<c2;
}
